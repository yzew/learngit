```c++
#include <vector>
#include <iostream>
using namespace std;

bool bisearch(const vector<int> &a, int target)
{
    int l = 0, r = a.size() - 1, mid;
    while (l <= r)
    {
        mid = (l + r) / 2;
        if (a[mid] == target)
            return true;
        else if (a[mid] > target)
            r = mid - 1;
        else
            l = mid + 1;
    }
    // if (a[l] == target || a[r] == target)
    //     return true;
    return false;
}

int main()
{
    vector<int> a(5);
    a[0] = 1, a[1] = 5, a[2] = 8, a[3] = 9, a[4] = 11;
    int target = 6;
    if (bisearch(a, target))
    {
        cout << "found " << target << "\n";
    }
    else
    {
        cout << target << " not found in array a\n";
    }
    return 0;
}
```
